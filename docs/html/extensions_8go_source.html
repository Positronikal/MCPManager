76// DiscoverFromExtensions discovers servers from Claude Extensions
77func (ced *ClaudeExtensionsDiscovery) DiscoverFromExtensions() ([]models.MCPServer, error) {
78 var allServers []models.MCPServer
79
80 // Get config directory
81 configDir := ced.pathResolver.GetConfigDir()
82 fmt.Printf(" Config directory: %s\n", configDir)
83 if configDir == "" {
84 return allServers, fmt.Errorf("could not determine config directory")
85 }
86
87 // Define paths
88 extensionsDir := filepath.Join(configDir, "Claude", "Claude Extensions")
89 settingsDir := filepath.Join(configDir, "Claude", "Claude Extensions Settings")
90
91 fmt.Printf(" Extensions directory: %s\n", extensionsDir)
92 fmt.Printf(" Settings directory: %s\n", settingsDir)
93
94 // Check if extensions directory exists
95 if _, err := os.Stat(extensionsDir); os.IsNotExist(err) {
96 fmt.Printf(" Extensions directory does not exist\n")
97 return allServers, nil
98 }
99
100 // Scan extensions directory
101 entries, err := os.ReadDir(extensionsDir)
102 if err != nil {
103 return allServers, fmt.Errorf("failed to read extensions directory: %w", err)
104 }
105
106 fmt.Printf(" Found %d extension entries\n", len(entries))
107
108 // Process each extension
109 for _, entry := range entries {
110 if !entry.IsDir() {
111 continue
112 }
113
114 extensionID := entry.Name()
115 fmt.Printf(" Scanning extension: %s\n", extensionID)
116
117 // Read manifest
118 manifestPath := filepath.Join(extensionsDir, extensionID, "manifest.json")
119 manifest, err := ced.readManifest(manifestPath)
120 if err != nil {
121 fmt.Printf(" ERROR reading manifest: %v\n", err)
122 continue
123 }
124
125 fmt.Printf(" Found: %s v%s\n", manifest.DisplayName, manifest.Version)
126
127 // Check if extension has an MCP server
128 if manifest.Server.MCPConfig.Command == "" {
129 fmt.Printf(" No MCP server configuration found\n")
130 continue
131 }
132
133 // Read settings
134 settingsPath := filepath.Join(settingsDir, extensionID+".json")
135 settings, err := ced.readSettings(settingsPath)
136 if err != nil {
137 fmt.Printf(" WARNING: Could not read settings: %v (treating as enabled)\n", err)
138 // Default to enabled if settings don't exist
139 settings = &ExtensionSettings{IsEnabled: true}
140 }
141
142 fmt.Printf(" Extension enabled: %v\n", settings.IsEnabled)
143
144 // Skip disabled extensions
145 if !settings.IsEnabled {
146 fmt.Printf(" Skipped (disabled)\n")
147 continue
148 }
149
150 // Create server model
151 extensionPath := filepath.Join(extensionsDir, extensionID)
152 server := ced.createServerFromExtension(manifest, settings, extensionPath, extensionID)
153
154 fmt.Printf(" Added to server list (ID: %s)\n", server.ID)
155 allServers = append(allServers, *server)
156 }
157
158 // Publish discovery events
159 for i := range allServers {
160 if ced.eventBus != nil {
161 ced.eventBus.Publish(events.ServerDiscoveredEvent(&allServers[i]))
162 }
163 }
164
165 return allServers, nil
166}

204// createServerFromExtension creates an MCPServer model from an extension
205func (ced *ClaudeExtensionsDiscovery) createServerFromExtension(
206 manifest *ExtensionManifest,
207 settings *ExtensionSettings,
208 extensionPath string,
209 extensionID string,
210) *models.MCPServer {
211 // Use display name as server name
212 serverName := manifest.DisplayName
213 if serverName == "" {
214 serverName = manifest.Name
215 }
216 if serverName == "" {
217 serverName = extensionID
218 }
219
220 // Resolve command and args
221 command := manifest.Server.MCPConfig.Command
222 args := ced.resolveArgs(manifest.Server.MCPConfig.Args, extensionPath, settings)
223
224 // Create server with extension discovery source
225 server := models.NewMCPServer(serverName, command, models.DiscoveryExtension)
226
227 // Set version
228 server.Version = manifest.Version
229
230 // Set configuration
231 server.Configuration.CommandLineArguments = args
232 server.Configuration.EnvironmentVariables = manifest.Server.MCPConfig.Env
233
234 // Store extension metadata in configuration
235 if server.Configuration.EnvironmentVariables == nil {
236 server.Configuration.EnvironmentVariables = make(map[string]string)
237 }
238 server.Configuration.EnvironmentVariables["__EXTENSION_ID__"] = extensionID
239 server.Configuration.EnvironmentVariables["__EXTENSION_PATH__"] = extensionPath
240 server.Configuration.EnvironmentVariables["__EXTENSION_TYPE__"] = manifest.Server.Type
241
242 // Extract tools/capabilities
243 server.Capabilities = make([]string, 0, len(manifest.Tools))
244 server.Tools = make([]string, 0, len(manifest.Tools))
245 for _, tool := range manifest.Tools {
246 server.Tools = append(server.Tools, tool.Name)
247 server.Capabilities = append(server.Capabilities, tool.Name)
248 }
249
250 // Detect transport type
251 server.Transport = ced.detectTransport(manifest, command)
252
253 return server
254}

256// detectTransport determines the transport type for a server
257func (ced *ClaudeExtensionsDiscovery) detectTransport(manifest *ExtensionManifest, command string) models.TransportType {
258 // Check if transport is explicitly defined in manifest
259 // (Future: MCP spec may add transport field to manifest)
260
261 // For now, use heuristics based on command and server type
262 cmdLower := strings.ToLower(command)
263
264 // Node.js servers typically use stdio
265 if cmdLower == "node" || strings.Contains(cmdLower, "node") {
266 return models.TransportStdio
267 }
268
269 // Python/UV servers typically use stdio
270 if cmdLower == "python" || cmdLower == "python3" || cmdLower == "uv" {
271 return models.TransportStdio
272 }
273
274 // If server type hints at HTTP/SSE
275 serverType := strings.ToLower(manifest.Server.Type)
276 if strings.Contains(serverType, "http") {
277 return models.TransportHTTP
278 }
279 if strings.Contains(serverType, "sse") {
280 return models.TransportSSE
281 }
282
283 // Default to stdio for most MCP servers
284 return models.TransportStdio
285}

287// resolveArgs resolves template variables in args like ${__dirname} and ${user_config.*}
288func (ced *ClaudeExtensionsDiscovery) resolveArgs(args []string, extensionPath string, settings *ExtensionSettings) []string {
289 resolved := make([]string, 0, len(args))
290
291 for _, arg := range args {
292 // Replace ${__dirname} with extension path
293 resolvedArg := strings.ReplaceAll(arg, "${__dirname}", extensionPath)
294
295 // Replace user_config variables
296 // Example: ${user_config.allowed_directories}
297 if strings.Contains(resolvedArg, "${user_config.") {
298 // Extract config key
299 start := strings.Index(resolvedArg, "${user_config.") + 14
300 end := strings.Index(resolvedArg[start:], "}")
301 if end > 0 {
302 configKey := resolvedArg[start : start+end]
303
304 // Look up in settings.UserConfig
305 if settings != nil && settings.UserConfig != nil {
306 if value, ok := settings.UserConfig[configKey]; ok {
307 // Check if value is an array - expand into multiple args
308 if arrayValue, isArray := value.([]interface{}); isArray {
309 // Array values expand into multiple arguments
310 for _, item := range arrayValue {
311 if str, ok := item.(string); ok {
312 resolved = append(resolved, str)
313 }
314 }
315 // Skip appending the template arg itself
316 continue
317 } else {
318 // Non-array value - convert to string
319 resolvedArg = ced.userConfigValueToString(value)
320 }
321 } else {
322 // Config key not found, skip this arg
323 continue
324 }
325 } else {
326 // No user config, skip this arg
327 continue
328 }
329 }
330 }
331
332 resolved = append(resolved, resolvedArg)
333 }
334
335 return resolved
336}

338// userConfigValueToString converts a user config value to a string
339// Note: Arrays are handled separately in resolveArgs() by expanding to multiple args
340func (ced *ClaudeExtensionsDiscovery) userConfigValueToString(value interface{}) string {
341 switch v := value.(type) {
342 case string:
343 return v
344 case []interface{}:
345 // Arrays should not reach here - they're expanded in resolveArgs()
346 // But if they do, join with commas as fallback
347 strValues := make([]string, 0, len(v))
348 for _, item := range v {
349 if str, ok := item.(string); ok {
350 strValues = append(strValues, str)
351 }
352 }
353 return strings.Join(strValues, ",")
354 case map[string]interface{}:
355 // Object - serialize as JSON (best effort)
356 if data, err := json.Marshal(v); err == nil {
357 return string(data)
358 }
359 return ""
360 default:
361 return fmt.Sprintf("%v", v)
362 }
363}
