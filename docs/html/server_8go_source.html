41// GenerateDeterministicUUID creates a stable UUID based on server identity
42// This ensures the same server always gets the same UUID across discoveries
43func GenerateDeterministicUUID(name, installationPath string, source DiscoverySource) string {
44 // Create a unique identifier string combining:
45 // - Server name (e.g., "Filesystem")
46 // - Installation path/command (e.g., "node")
47 // - Source (e.g., "extension")
48 // This ensures the same server always gets the same ID
49 identityString := fmt.Sprintf("%s|%s|%s", name, installationPath, source)
50
51 // Hash the identity string to create a deterministic but unique value
52 hash := sha256.Sum256([]byte(identityString))
53
54 // Convert first 16 bytes of hash to UUID format
55 // This creates a valid UUID v5-style deterministic ID
56 hashHex := hex.EncodeToString(hash[:16])
57
58 // Format as UUID: xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx
59 uuidStr := fmt.Sprintf("%s-%s-%s-%s-%s",
60 hashHex[0:8],
61 hashHex[8:12],
62 hashHex[12:16],
63 hashHex[16:20],
64 hashHex[20:32],
65 )
66
67 return uuidStr
68}

90// Validate checks if the MCPServer is in a valid state
91func (s *MCPServer) Validate() error {
92 // Validate ID is a valid UUID
93 if _, err := uuid.Parse(s.ID); err != nil {
94 return fmt.Errorf("invalid server ID (must be UUID): %s", s.ID)
95 }
96
97 // Validate name is not empty
98 if s.Name == "" {
99 return fmt.Errorf("server name cannot be empty")
100 }
101
102 // Validate installation path is not empty
103 if s.InstallationPath == "" {
104 return fmt.Errorf("installation path cannot be empty")
105 }
106
107 // Check if installation path exists
108 if _, err := os.Stat(s.InstallationPath); err != nil {
109 if os.IsNotExist(err) {
110 return fmt.Errorf("installation path does not exist: %s", s.InstallationPath)
111 }
112 return fmt.Errorf("cannot access installation path: %w", err)
113 }
114
115 // Validate PID consistency with status
116 if s.Status.State == StatusRunning || s.Status.State == StatusStarting {
117 if s.PID == nil {
118 return fmt.Errorf("PID must be set when status is %s", s.Status.State)
119 }
120 if *s.PID <= 0 {
121 return fmt.Errorf("PID must be positive, got: %d", *s.PID)
122 }
123 } else if s.Status.State == StatusStopped || s.Status.State == StatusError {
124 // Note: PID can be set in error state (crashed process)
125 // but must be nil in stopped state
126 if s.Status.State == StatusStopped && s.PID != nil {
127 return fmt.Errorf("PID must be nil when status is stopped")
128 }
129 }
130
131 // Validate timestamp consistency
132 if s.LastSeenAt.Before(s.DiscoveredAt) {
133 return fmt.Errorf("lastSeenAt (%s) cannot be before discoveredAt (%s)",
134 s.LastSeenAt.Format(time.RFC3339), s.DiscoveredAt.Format(time.RFC3339))
135 }
136
137 // Validate discovery source
138 if !s.Source.IsValid() {
139 return fmt.Errorf("invalid discovery source: %s", s.Source)
140 }
141
142 // Validate nested structures
143 if err := s.Status.Validate(); err != nil {
144 return fmt.Errorf("invalid status: %w", err)
145 }
146
147 if err := s.Configuration.Validate(); err != nil {
148 return fmt.Errorf("invalid configuration: %w", err)
149 }
150
151 for i, dep := range s.Dependencies {
152 if err := dep.Validate(); err != nil {
153 return fmt.Errorf("invalid dependency at index %d: %w", i, err)
154 }
155 }
156
157 return nil
158}
