97// listProcessesWindows enumerates all processes using native Win32 API
98func (pd *ProcessDiscovery) listProcessesWindows() ([]ProcessInfo, error) {
99 // First, get snapshot of all processes to build parent-child relationships
100 snapshot, err := createProcessSnapshot()
101 if err != nil {
102 return nil, fmt.Errorf("failed to create process snapshot: %w", err)
103 }
104 defer windows.CloseHandle(windows.Handle(snapshot))
105
106 var processes []ProcessInfo
107
108 // Get first process
109 var pe PROCESSENTRY32
110 pe.Size = uint32(unsafe.Sizeof(pe))
111
112 ret, _, _ := procProcess32First.Call(uintptr(snapshot), uintptr(unsafe.Pointer(&pe)))
113 if ret == 0 {
114 return nil, fmt.Errorf("Process32First failed")
115 }
116
117 // Iterate through all processes
118 for {
119 pid := pe.ProcessID
120 parentPID := pe.ParentProcessID
121 exeName := windows.UTF16ToString(pe.ExeFile[:])
122
123 // Skip system idle process
124 if pid == 0 {
125 goto next
126 }
127
128 // Filter to relevant processes for MCP servers
129 // We're interested in: node.exe, python.exe, uv.exe, go.exe, claude.exe
130 if !isRelevantProcess(exeName) {
131 goto next
132 }
133
134 // Try to get more detailed information
135 {
136 procInfo, err := getProcessDetails(pid, parentPID, exeName)
137 if err != nil {
138 // If we can't get details, at least add basic info
139 processes = append(processes, ProcessInfo{
140 PID: int(pid),
141 ParentPID: int(parentPID),
142 Name: exeName,
143 CommandLine: exeName,
144 })
145 } else {
146 processes = append(processes, procInfo)
147 }
148 }
149
150 next:
151 // Get next process
152 ret, _, _ = procProcess32Next.Call(uintptr(snapshot), uintptr(unsafe.Pointer(&pe)))
153 if ret == 0 {
154 break
155 }
156 }
157
158 return processes, nil
159}

173// isRelevantProcess checks if a process is relevant for MCP server detection
174func isRelevantProcess(name string) bool {
175 relevantNames := []string{
176 "node.exe",
177 "python.exe",
178 "python3.exe",
179 "uv.exe",
180 "go.exe",
181 "claude.exe",
182 }
183
184 nameLower := toLower(name)
185 for _, relevant := range relevantNames {
186 if nameLower == toLower(relevant) {
187 return true
188 }
189 }
190 return false
191}

193// getProcessDetails gets detailed information about a process including command line
194func getProcessDetails(pid, parentPID uint32, exeName string) (ProcessInfo, error) {
195 // Open process with query and read permissions
196 handle, err := windows.OpenProcess(
197 PROCESS_QUERY_INFORMATION|PROCESS_VM_READ,
198 false,
199 pid,
200 )
201 if err != nil {
202 return ProcessInfo{}, fmt.Errorf("OpenProcess failed: %w", err)
203 }
204 defer windows.CloseHandle(handle)
205
206 // Get executable path
207 var exePath [MAX_PATH]uint16
208 size := uint32(MAX_PATH)
209 ret, _, _ := procGetModuleFileNameEx.Call(
210 uintptr(handle),
211 0,
212 uintptr(unsafe.Pointer(&exePath[0])),
213 uintptr(size),
214 )
215
216 executablePath := exeName
217 if ret > 0 {
218 executablePath = windows.UTF16ToString(exePath[:])
219 }
220
221 // Get command line from PEB
222 commandLine, err := getProcessCommandLine(handle)
223 if err != nil {
224 // If we can't get command line, use executable path as fallback
225 commandLine = executablePath
226 }
227
228 return ProcessInfo{
229 PID: int(pid),
230 ParentPID: int(parentPID),
231 Name: exeName,
232 CommandLine: commandLine,
233 }, nil
234}

236// getProcessCommandLine extracts the full command line from a process's PEB
237func getProcessCommandLine(handle windows.Handle) (string, error) {
238 // Query process basic information to get PEB address
239 var pbi PROCESS_BASIC_INFORMATION
240 var returnLength uint32
241
242 ret, _, _ := procNtQueryInformationProcess.Call(
243 uintptr(handle),
244 0, // ProcessBasicInformation
245 uintptr(unsafe.Pointer(&pbi)),
246 uintptr(unsafe.Sizeof(pbi)),
247 uintptr(unsafe.Pointer(&returnLength)),
248 )
249
250 if ret != 0 {
251 return "", fmt.Errorf("NtQueryInformationProcess failed with status: 0x%x", ret)
252 }
253
254 // Read PEB structure from process memory
255 var peb PEB
256 err := readProcessMemory(handle, pbi.PebBaseAddress, unsafe.Pointer(&peb), unsafe.Sizeof(peb))
257 if err != nil {
258 return "", fmt.Errorf("failed to read PEB: %w", err)
259 }
260
261 // Read RTL_USER_PROCESS_PARAMETERS structure
262 var params RTL_USER_PROCESS_PARAMETERS
263 err = readProcessMemory(handle, peb.ProcessParameters, unsafe.Pointer(&params), unsafe.Sizeof(params))
264 if err != nil {
265 return "", fmt.Errorf("failed to read process parameters: %w", err)
266 }
267
268 // Read the actual command line string
269 if params.CommandLine.Length == 0 || params.CommandLine.Buffer == 0 {
270 return "", fmt.Errorf("command line is empty")
271 }
272
273 // Allocate buffer for command line
274 cmdLineLength := params.CommandLine.Length / 2 // Length is in bytes, we need characters
275 if cmdLineLength > 32768 { // Sanity check
276 cmdLineLength = 32768
277 }
278
279 cmdLineBuffer := make([]uint16, cmdLineLength+1)
280 err = readProcessMemory(
281 handle,
282 params.CommandLine.Buffer,
283 unsafe.Pointer(&cmdLineBuffer[0]),
284 uintptr(params.CommandLine.Length),
285 )
286 if err != nil {
287 return "", fmt.Errorf("failed to read command line string: %w", err)
288 }
289
290 return windows.UTF16ToString(cmdLineBuffer), nil
291}

316func findClaudeDesktopChildren() ([]uint32, error) {
317 // First, find Claude.exe PID
318 snapshot, err := createProcessSnapshot()
319 if err != nil {
320 return nil, err
321 }
322 defer windows.CloseHandle(windows.Handle(snapshot))
323

325 var pe PROCESSENTRY32
326 pe.Size = uint32(unsafe.Sizeof(pe))
327
328 ret, _, _ := procProcess32First.Call(uintptr(snapshot), uintptr(unsafe.Pointer(&pe)))
329 if ret == 0 {
330 return nil, fmt.Errorf("Process32First failed")
331 }
332
333 // Find Claude.exe
334 for {
335 exeName := windows.UTF16ToString(pe.ExeFile[:])
336 if toLower(exeName) == "claude.exe" {
337 claudePID = pe.ProcessID
338 break
339 }
340
341 ret, _, _ = procProcess32Next.Call(uintptr(snapshot), uintptr(unsafe.Pointer(&pe)))
342 if ret == 0 {
343 break
344 }
345 }

348 return nil, fmt.Errorf("Claude.exe not found")
349 }

352 var children []uint32
353
354 // Reset snapshot
355 snapshot2, err := createProcessSnapshot()
356 if err != nil {
357 return nil, err
358 }

360
361 pe.Size = uint32(unsafe.Sizeof(pe))

363 if ret == 0 {

366

381// toLower is a simple ASCII lowercase function to avoid string imports
382func toLower(s string) string {
383 b := make([]byte, len(s))
384 for i := 0; i < len(s); i++ {
385 c := s[i]
386 if c >= 'A' && c <= 'Z' {
387 c = c + ('a' - 'A')
388 }
389 b[i] = c
390 }
391 return string(b)
392}
