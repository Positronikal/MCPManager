71// parsePsOutput parses Unix ps aux output
72func (pd *ProcessDiscovery) parsePsOutput(output string) ([]ProcessInfo, error) {
73 var processes []ProcessInfo
74
75 lines := strings.Split(output, "\n")
76 for i, line := range lines {
77 if i == 0 || line == "" {
78 continue // Skip header
79 }
80
81 fields := strings.Fields(line)
82 if len(fields) < 11 {
83 continue
84 }
85
86 pidStr := fields[1]
87 pid, err := strconv.Atoi(pidStr)
88 if err != nil {
89 continue
90 }
91
92 // Command line starts at field 10
93 commandLine := strings.Join(fields[10:], " ")
94 name := fields[10]
95
96 processes = append(processes, ProcessInfo{
97 PID: pid,
98 Name: name,
99 CommandLine: commandLine,
100 })
101 }
102
103 return processes, nil
104}

106// parseCSVLine parses a CSV line respecting quoted fields
107func parseCSVLine(line string) []string {
