27// State transition rules based on the state machine diagram:
28// stopped -> starting (user initiates start)
29// starting -> running (startup successful)
30// starting -> error (startup failed)
31// starting -> stopped (user cancels startup)
32// running -> stopped (user initiates stop)
33// running -> error (crash/unexpected termination)
34// error -> starting (retry attempt)
35// error -> stopped (user cancels retry)
36
37var validTransitions = map[StatusState][]StatusState{
38 StatusStopped: {StatusStarting},
39 StatusStarting: {StatusRunning, StatusError, StatusStopped}, // Allow canceling startup
40 StatusRunning: {StatusStopped, StatusError},
41 StatusError: {StatusStarting, StatusStopped},
42}

44// CanTransitionTo checks if transitioning to the new state is valid
45func (s *ServerStatus) CanTransitionTo(newState StatusState) bool {
46 if !newState.IsValid() {
47 return false
48 }
49
50 allowedStates, exists := validTransitions[s.State]
51 if !exists {
52 return false
53 }
54
55 for _, allowed := range allowedStates {
56 if allowed == newState {
57 return true
58 }
59 }
60
61 return false
62}

64// TransitionTo attempts to transition to a new state
65func (s *ServerStatus) TransitionTo(newState StatusState, reason string) error {
66 if !s.CanTransitionTo(newState) {
67 return fmt.Errorf("invalid state transition from %s to %s", s.State, newState)
68 }
69
70 oldState := s.State
71 s.State = newState
72 s.LastStateChange = time.Now()
73
74 // Handle startup attempts logic
75 if newState == StatusStarting {
76 // Increment startup attempts when entering starting state
77 s.StartupAttempts++
78 } else if newState == StatusRunning && oldState == StatusStarting {
79 // Reset startup attempts on successful start
80 s.StartupAttempts = 0
81 s.ErrorMessage = ""
82 } else if newState == StatusError {
83 // Set error message
84 s.ErrorMessage = reason
85 // Determine if crash is recoverable
86 s.CrashRecoverable = s.StartupAttempts < 3
87 } else if newState == StatusStopped {
88 // Reset error state when stopped
89 s.ErrorMessage = ""
90 }
91
92 return nil
93}

95// Validate checks if the ServerStatus is in a valid state
96func (s *ServerStatus) Validate() error {
97 if !s.State.IsValid() {

100
101 if s.StartupAttempts < 0 {
102 return fmt.Errorf("startupAttempts cannot be negative: %d", s.StartupAttempts)
103 }
104
105 if s.State == StatusError && s.ErrorMessage == "" {
106 return fmt.Errorf("error state requires an error message")
107 }
108
109 return nil
