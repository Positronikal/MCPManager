76// ReadConfig reads and parses an MCP client configuration file
77func (ce *ClientEditor) ReadConfig(configPath string) (*ClientConfig, error) {
78 // Check if file exists
79 if !ce.fileExists(configPath) {
80 // Return empty config if file doesn't exist
81 return &ClientConfig{
82 MCPServers: make(map[string]ServerEntry),
83 }, nil
84 }
85
86 // Read file
87 data, err := os.ReadFile(configPath)
88 if err != nil {
89 return nil, fmt.Errorf("failed to read config file: %w", err)
90 }
91
92 // Parse JSON
93 var config ClientConfig
94 if err := json.Unmarshal(data, &config); err != nil {
95 return nil, fmt.Errorf("failed to parse config file: %w", err)
96 }
97
98 // Ensure MCPServers map is initialized
99 if config.MCPServers == nil {
100 config.MCPServers = make(map[string]ServerEntry)
101 }
102
103 return &config, nil
104}

106// WriteConfig writes an updated configuration to the client config file
107// Creates a backup before writing and validates the JSON structure
108func (ce *ClientEditor) WriteConfig(configPath string, config *ClientConfig) error {
109 // Validate config structure
110 if config == nil {
111 return fmt.Errorf("config cannot be nil")
112 }
113 if config.MCPServers == nil {
114 config.MCPServers = make(map[string]ServerEntry)
115 }
116
117 // Create backup if file exists
118 if ce.fileExists(configPath) {
119 if err := ce.createBackup(configPath); err != nil {
120 return fmt.Errorf("failed to create backup: %w", err)
121 }
122 }
123
124 // Ensure directory exists
125 dir := filepath.Dir(configPath)
126 if err := os.MkdirAll(dir, 0755); err != nil {
127 return fmt.Errorf("failed to create config directory: %w", err)
128 }
129
130 // Marshal config to JSON with indentation for readability
131 data, err := json.MarshalIndent(config, "", " ")
132 if err != nil {
133 return fmt.Errorf("failed to marshal config: %w", err)
134 }
135
136 // Write to file
137 if err := os.WriteFile(configPath, data, 0644); err != nil {
138 return fmt.Errorf("failed to write config file: %w", err)
139 }
140
141 return nil
142}

212// getClaudeDesktopConfigPath returns the path to the Claude Desktop config file
213func (ce *ClientEditor) getClaudeDesktopConfigPath() string {
214 // Windows: %APPDATA%\Claude\claude_desktop_config.json
215 // macOS: ~/Library/Application Support/Claude/claude_desktop_config.json
216 // Linux: ~/.config/Claude/claude_desktop_config.json
217
218 homeDir, err := os.UserHomeDir()
219 if err != nil {
220 return ""
221 }
222
223 // Detect OS
224 switch {
225 case fileExists(filepath.Join(os.Getenv("APPDATA"), "Claude")):
226 // Windows
227 return filepath.Join(os.Getenv("APPDATA"), "Claude", "claude_desktop_config.json")
228 case fileExists(filepath.Join(homeDir, "Library", "Application Support", "Claude")):
229 // macOS
230 return filepath.Join(homeDir, "Library", "Application Support", "Claude", "claude_desktop_config.json")
231 default:
232 // Linux/Unix
233 return filepath.Join(homeDir, ".config", "Claude", "claude_desktop_config.json")
234 }
235}
