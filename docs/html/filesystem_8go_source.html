73// discoverNPMServers discovers MCP servers from NPM global packages
74func (fd *FilesystemDiscovery) discoverNPMServers() ([]models.MCPServer, error) {
75 var servers []models.MCPServer
76
77 // Get NPM global root
78 fmt.Println(" Running: npm root -g")
79 cmd := exec.Command("npm", "root", "-g")
80 output, err := cmd.Output()
81 if err != nil {
82 // NPM not installed or not in PATH
83 return servers, err
84 }
85
86 npmRoot := strings.TrimSpace(string(output))
87 fmt.Printf(" NPM global root: %s\n", npmRoot)
88 if npmRoot == "" {
89 return servers, fmt.Errorf("npm root returned empty")
90 }
91
92 // Check if directory exists
93 if _, err := os.Stat(npmRoot); os.IsNotExist(err) {
94 fmt.Printf(" Directory does not exist\n")
95 return servers, nil
96 }
97
98 // Scan for MCP server packages
99 entries, err := os.ReadDir(npmRoot)
100 if err != nil {
101 return servers, err
102 }
103
104 fmt.Printf(" Scanning %d entries in NPM directory...\n", len(entries))
105 for _, entry := range entries {
106 if !entry.IsDir() {
107 continue
108 }
109
110 name := entry.Name()
111
112 // Match patterns for MCP servers
113 if isMCPServerPackage(name) {
114 fmt.Printf(" Found MCP server package: %s\n", name)
115 serverPath := filepath.Join(npmRoot, name)
116
117 // Try to find the main entry point
118 packageJSONPath := filepath.Join(serverPath, "package.json")
119 if _, err := os.Stat(packageJSONPath); err == nil {
120 // Create server entry
121 server := models.NewMCPServer(name, serverPath, models.DiscoveryFilesystem)
122 server.Configuration.CommandLineArguments = []string{}
123 servers = append(servers, *server)
124 }
125 }
126 }
127
128 return servers, nil
129}

234// isMCPServerPackage determines if a package name is likely an MCP server
235// This uses conservative patterns to avoid false positives
236func isMCPServerPackage(name string) bool {
237 nameLower := strings.ToLower(name)
238
239 // Official MCP servers from @modelcontextprotocol scope
240 if strings.HasPrefix(nameLower, "@modelcontextprotocol/") ||
241 strings.HasPrefix(nameLower, "modelcontextprotocol") {
242 return true
243 }
244
245 // Common MCP server naming patterns (but not just "mcp" alone)
246 // Use hyphens or underscores to avoid matching "mcpmanager"
247 if strings.Contains(nameLower, "mcp-server") ||
248 strings.Contains(nameLower, "mcp_server") ||
249 strings.Contains(nameLower, "-mcp-") ||
250 strings.Contains(nameLower, "_mcp_") {
251 return true
252 }
253
254 // Specific known MCP servers (add more as discovered)
255 knownServers := []string{
256 "figma-mcp",
257 "windows-mcp",
258 "slack-mcp",
259 "github-mcp",
260 }
261
262 for _, known := range knownServers {
263 if strings.Contains(nameLower, known) {
264 return true
265 }
266 }
267
268 return false
269}
