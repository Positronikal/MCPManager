40// GetMetrics retrieves current metrics for a server
41// Returns metrics for running servers, nil metrics for stopped servers
42// Rate-limited to 1Hz per server as per research.md ยง6
43func (mc *MetricsCollector) GetMetrics(serverID string, status *models.ServerStatus, pid int) (*models.ServerMetrics, error) {
44 // Check if server is running
45 if status.State != models.StatusRunning {
46 // Return empty metrics for non-running servers
47 metrics := models.NewServerMetrics(serverID)
48 return metrics, nil
49 }
50
51 // Check rate limit (1Hz = 1 update per second)
52 if !mc.shouldUpdate(serverID) {
53 // Return cached metrics if within rate limit
54 return mc.getCachedMetrics(serverID), nil
55 }
56
57 // Collect fresh metrics
58 metrics := models.NewServerMetrics(serverID)
59
60 // Calculate uptime from LastStateChange
61 if status.State == models.StatusRunning {
62 metrics.Uptime = time.Since(status.LastStateChange)
63 }
64
65 // Get memory usage if PID is available
66 if pid > 0 {
67 memBytes, err := mc.processInfo.GetMemoryUsage(pid)
68 if err == nil && memBytes > 0 {
69 metrics.MemoryBytes = &memBytes
70 }
71 // Silently ignore errors - memory might not be available
72 }
73
74 // Request count is not implemented yet (would require MCP protocol support)
75 // Set to nil for now
76
77 // Cache the metrics
78 mc.cacheMetrics(serverID, metrics, pid, status.LastStateChange)
79
80 // Update rate limit timestamp
81 mc.updateRateLimit(serverID)
82
83 // Publish metrics updated event
84 if mc.eventBus != nil {
85 event := events.ServerMetricsUpdatedEvent(serverID, map[string]interface{}{
86 "uptime": metrics.Uptime.Seconds(),
87 "memoryBytes": metrics.MemoryBytes,
88 })
89 mc.eventBus.Publish(event)
90 }
91
92 return metrics, nil
93}

95// shouldUpdate checks if we should update metrics based on rate limit
96func (mc *MetricsCollector) shouldUpdate(serverID string) bool {
97 mc.rateLimitMu.RLock()
98 defer mc.rateLimitMu.RUnlock()
99
100 lastUpdate, exists := mc.rateLimitCache[serverID]
101 if !exists {
102 return true
103 }
104
105 // Allow update if more than 1 second has passed
106 return time.Since(lastUpdate) >= time.Second
107}

129// getCachedMetrics retrieves cached metrics for a server

131 mc.mu.RLock()
132 defer mc.mu.RUnlock()
133
