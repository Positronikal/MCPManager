40// SSEStream handles GET /api/v1/events
41func (h *SSEHandlers) SSEStream(w http.ResponseWriter, r *http.Request) {
42 // Set SSE headers
43 w.Header().Set("Content-Type", "text/event-stream")
44 w.Header().Set("Cache-Control", "no-cache")
45 w.Header().Set("Connection", "keep-alive")
46 w.Header().Set("Access-Control-Allow-Origin", "*")
47
48 // Parse query parameters
49 serverIDsParam := r.URL.Query().Get("serverIds")
50
51 // Parse server IDs filter
52 serverIDsFilter := make(map[string]bool)
53 if serverIDsParam != "" {
54 ids := strings.Split(serverIDsParam, ",")
55 for _, id := range ids {
56 trimmedID := strings.TrimSpace(id)
57 if _, err := uuid.Parse(trimmedID); err == nil {
58 serverIDsFilter[trimmedID] = true
59 }
60 }
61 }
62
63 // Create subscription
64 ctx, cancel := context.WithCancel(r.Context())
65 sub := &sseSubscription{
66 id: uuid.New().String(),
67 serverIDs: serverIDsFilter,
68 cancelFunc: cancel,
69 connectedAt: time.Now(),
70 lastEventID: 0,
71 }
72
73 // Register subscription
74 h.subscriptionsMu.Lock()
75 h.subscriptions[sub.id] = sub
76 h.subscriptionsMu.Unlock()
77
78 // Cleanup on disconnect
79 defer func() {
80 h.subscriptionsMu.Lock()
81 delete(h.subscriptions, sub.id)
82 h.subscriptionsMu.Unlock()
83 cancel()
84 }()
85
86 // Subscribe to all event types from EventBus
87 eventTypes := []events.EventType{
88 events.EventServerDiscovered,
89 events.EventServerStatusChanged,
90 events.EventServerLogEntry,
91 events.EventConfigFileChanged,
92 events.EventServerMetricsUpdated,
93 }
94
95 // Create a combined channel for all events
96 combinedChan := make(chan *events.Event, 100)
97 var channels []<-chan *events.Event
98
99 // Subscribe to each event type
100 for _, eventType := range eventTypes {
101 ch := h.eventBus.Subscribe(eventType)
102 channels = append(channels, ch)
103 }
104
105 // Start goroutine to merge all channels into one
106 go func() {
107 for _, ch := range channels {
108 go func(eventChan <-chan *events.Event) {
109 for event := range eventChan {
110 if event != nil {
111 select {
112 case combinedChan <- event:
113 case <-ctx.Done():
114 return
115 }
116 }
117 }
118 }(ch)
119 }
120 }()
121
122 // Create heartbeat ticker
123 heartbeat := time.NewTicker(15 * time.Second)
124 defer heartbeat.Stop()
125
126 // Flush immediately to establish connection
127 if f, ok := w.(http.Flusher); ok {
128 f.Flush()
129 }
130
131 // Event loop
132 for {
133 select {
134 case <-ctx.Done():
135 // Client disconnected
136 return
137
138 case event := <-combinedChan:
139 // Filter event by server ID if needed
140 if !h.shouldSendEvent(sub, event) {
141 continue
142 }
143
144 // Write event
145 sub.lastEventID++
146 h.writeSSEEvent(w, event, sub.lastEventID)
147
148 // Flush
149 if f, ok := w.(http.Flusher); ok {
150 f.Flush()
151 }
152
153 case <-heartbeat.C:
154 // Send heartbeat comment
155 fmt.Fprintf(w, ": heartbeat\n\n")
156
157 // Flush
158 if f, ok := w.(http.Flusher); ok {
159 f.Flush()
160 }
161 }
162 }
163}

165// shouldSendEvent checks if an event should be sent to a subscription based on filters
166func (h *SSEHandlers) shouldSendEvent(sub *sseSubscription, event *events.Event) bool {
167 // If no server filter, send all events
168 if len(sub.serverIDs) == 0 {
169 return true
170 }
171
172 // Check if event's server ID matches filter
173 serverID, ok := event.Data["serverID"].(string)
174 if !ok || serverID == "" {
175 return true // Send events without server ID
176 }
177
178 return sub.serverIDs[serverID]
179}

181// writeSSEEvent writes an event in SSE format
182func (h *SSEHandlers) writeSSEEvent(w http.ResponseWriter, event *events.Event, eventID int) {
183 // SSE format:
184 // id: <event-id>
185 // event: <event-type>
186 // data: <json-payload>
187 // (blank line)
188
189 fmt.Fprintf(w, "id: %d\n", eventID)
190 fmt.Fprintf(w, "event: %s\n", event.Type)
191
192 // Marshal event data to JSON
193 dataJSON, err := json.Marshal(event)
194 if err != nil {
195 // Log error and send error event
196 fmt.Fprintf(w, "data: {\"error\": \"failed to marshal event\"}\n\n")
197 return
198 }
199
200 fmt.Fprintf(w, "data: %s\n\n", string(dataJSON))
201}
