53// StartServer handles POST /api/v1/servers/{serverId}/start
54func (h *LifecycleHandlers) StartServer(w http.ResponseWriter, r *http.Request) {
55 // Extract server ID from URL
56 serverID := chi.URLParam(r, "serverId")
57
58 // Validate UUID format
59 if _, err := uuid.Parse(serverID); err != nil {
60 respondError(w, http.StatusNotFound, "Invalid server ID format")
61 return
62 }
63
64 // Get server from discovery service
65 server, exists := h.discoveryService.GetServerByID(serverID)
66 if !exists {
67 respondError(w, http.StatusNotFound, "Server not found")
68 return
69 }
70
71 // Check if server is already running
72 if server.Status.State == models.StatusRunning || server.Status.State == models.StatusStarting {
73 respondError(w, http.StatusBadRequest, "Server is already running or starting")
74 return
75 }
76
77 // Start server asynchronously
78 go func() {
79 err := h.lifecycleService.StartServer(server)
80 if err == nil {
81 // Update server in cache
82 h.discoveryService.UpdateServer(server)
83 }
84 }()
85
86 // Return 202 Accepted immediately
87 response := StartServerResponse{
88 Message: "Server starting",
89 ServerID: serverID,
90 Status: "starting",
91 }
92
93 respondJSON(w, http.StatusAccepted, response)
94}
