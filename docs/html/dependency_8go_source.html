19// IsInstalled checks if the dependency is installed and satisfies the version requirement
20func (d *Dependency) IsInstalled() bool {
21 // If no detected version, not installed
22 if d.DetectedVersion == "" {
23 return false
24 }
25
26 // If no required version, any version is acceptable
27 if d.RequiredVersion == "" {
28 return true
29 }
30
31 // Try to parse as semver constraint
32 constraint, err := semver.NewConstraint(d.RequiredVersion)
33 if err != nil {
34 // If not a valid semver constraint, do string comparison
35 return strings.EqualFold(d.DetectedVersion, d.RequiredVersion)
36 }
37
38 // Parse detected version
39 version, err := semver.NewVersion(d.DetectedVersion)
40 if err != nil {
41 // If detected version is not valid semver, do string comparison
42 return strings.EqualFold(d.DetectedVersion, d.RequiredVersion)
43 }
44
45 // Check if version satisfies constraint
46 return constraint.Check(version)
47}
