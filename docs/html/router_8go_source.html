27// NewRouter creates and configures the Chi router with all endpoints
28func NewRouter(services *Services) *chi.Mux {
29 r := chi.NewRouter()
30
31 // Apply middleware
32 r.Use(LoggingMiddleware)
33 r.Use(ErrorHandlingMiddleware)
34 r.Use(CORSMiddleware)
35
36 // Create handler instances
37 discoveryHandlers := NewDiscoveryHandlers(services.DiscoveryService)
38 lifecycleHandlers := NewLifecycleHandlers(services.LifecycleService, services.DiscoveryService)
39 configHandlers := NewConfigHandlers(services.ConfigService, services.DiscoveryService)
40 monitoringHandlers := NewMonitoringHandlers(services.MonitoringService, services.MetricsCollector, services.DiscoveryService)
41 dependencyHandlers := NewDependencyHandlers(services.DependencyService, services.UpdateChecker, services.DiscoveryService)
42 appStateHandlers := NewAppStateHandlers(services.StorageService)
43 sseHandlers := NewSSEHandlers(services.EventBus)
44
45 // Define API routes
46 r.Route("/api/v1", func(r chi.Router) {
47 // Discovery endpoints
48 r.Get("/servers", discoveryHandlers.ListServers)
49 r.Post("/servers/discover", discoveryHandlers.DiscoverServers)
50 r.Get("/servers/{serverId}", discoveryHandlers.GetServerByID)
51
52 // Lifecycle endpoints
53 r.Post("/servers/{serverId}/start", lifecycleHandlers.StartServer)
54 r.Post("/servers/{serverId}/stop", lifecycleHandlers.StopServer)
55 r.Post("/servers/{serverId}/restart", lifecycleHandlers.RestartServer)
56 r.Get("/servers/{serverId}/status", lifecycleHandlers.GetServerStatus)
57
58 // Configuration endpoints
59 r.Get("/servers/{serverId}/configuration", configHandlers.GetConfiguration)
60 r.Put("/servers/{serverId}/configuration", configHandlers.UpdateConfiguration)
61
62 // Monitoring endpoints
63 r.Get("/servers/{serverId}/logs", monitoringHandlers.GetServerLogs)
64 r.Get("/logs", monitoringHandlers.GetAllLogs)
65 r.Get("/servers/{serverId}/metrics", monitoringHandlers.GetServerMetrics)
66 r.Get("/netstat", monitoringHandlers.GetNetstat)
67 r.Get("/services", monitoringHandlers.GetServices)
68
69 // Dependency endpoints
70 r.Get("/servers/{serverId}/dependencies", dependencyHandlers.GetServerDependencies)
71 r.Get("/servers/{serverId}/updates", dependencyHandlers.GetServerUpdates)
72
73 // Application state endpoints
74 r.Get("/application/state", appStateHandlers.GetApplicationState)
75 r.Put("/application/state", appStateHandlers.UpdateApplicationState)
76
77 // SSE events endpoint
78 r.Get("/events", sseHandlers.SSEStream)
79 })
80
81 return r
82}
