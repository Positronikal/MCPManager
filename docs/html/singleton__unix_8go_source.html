33// Acquire attempts to acquire the single instance lock
34func (u *UnixSingleInstance) Acquire() (bool, error) {
35 // Try to connect to existing socket
36 conn, err := net.Dial("unix", u.socketPath)
37 if err == nil {
38 // Successfully connected - another instance is running
39 conn.Close()
40
41 // Try to signal the existing instance to show its window
42 u.signalExistingInstance()
43
44 return false, nil
45 }
46
47 // Socket doesn't exist or connection failed - try to create it
48 // First, clean up any stale socket file
49 if err := u.cleanupStaleSocket(); err != nil {
50 return false, fmt.Errorf("failed to cleanup stale socket: %w", err)
51 }
52
53 // Create the Unix domain socket listener
54 listener, err := net.Listen("unix", u.socketPath)
55 if err != nil {
56 return false, fmt.Errorf("failed to create socket listener: %w", err)
57 }
58
59 u.listener = listener
60
61 // Start a goroutine to accept connections (for future instance detection)
62 go u.acceptConnections()
63
64 return true, nil
65}

83func (u *UnixSingleInstance) cleanupStaleSocket() error {
84 // Check if socket file exists

86 return nil // No socket file, nothing to clean up
87 }
88
89 // Try to connect to the socket

91 if err == nil {
92 // Connection successful - another instance is actually running
93 conn.Close()
