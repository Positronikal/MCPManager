193export const filteredServers = derived(
194 [servers, serverFilters],
195 ([$servers, $filters]) => {
196 let result = $servers;
197
198 // Filter by status

250export function addNotification(type: Notification['type'], message: string, duration = 5000) {
251 const id = crypto.randomUUID();
252 const notification: Notification = {

262 // Auto-remove after duration

308export function addLog(serverId: string, log: LogEntry) {
309 // Update per-server logs
310 serverLogs.update(logsMap => {
311 if (!logsMap[serverId]) {
312 logsMap[serverId] = [];
313 }
314 logsMap[serverId].push(log);
315 // Keep only last 1000 logs per server
316 if (logsMap[serverId].length > 1000) {
317 logsMap[serverId] = logsMap[serverId].slice(-1000);
318 }
319 return logsMap;
320 });
321
322 // Update global logs store
323 logs.update(globalLogs => {
324 const updatedLogs = [...globalLogs, log];
325 // Keep only last 1000 logs globally
326 if (updatedLogs.length > 1000) {
327 return updatedLogs.slice(-1000);
328 }
329 return updatedLogs;
330 });
331}
