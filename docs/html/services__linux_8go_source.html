33// parseServicesLinux parses Linux systemctl output
34func parseServicesLinux(output string) ([]Service, error) {
35 var services []Service
36 lines := strings.Split(output, "\n")
37
38 for _, line := range lines {
39 line = strings.TrimSpace(line)
40 if line == "" {
41 continue
42 }
43
44 // Parse line format: "UNIT LOAD ACTIVE SUB DESCRIPTION"
45 // Example: "ssh.service loaded active running OpenSSH server daemon"
46 fields := strings.Fields(line)
47 if len(fields) < 4 {
48 continue
49 }
50
51 serviceName := fields[0]
52 // load := fields[1]
53 active := fields[2]
54 sub := fields[3]
55 description := ""
56 if len(fields) > 4 {
57 description = strings.Join(fields[4:], " ")
58 }
59
60 // Determine status from active and sub fields
61 status := "UNKNOWN"
62 if active == "active" {
63 if sub == "running" {
64 status = "RUNNING"
65 } else if sub == "exited" {
66 status = "EXITED"
67 } else {
68 status = strings.ToUpper(sub)
69 }
70 } else if active == "inactive" {
71 status = "STOPPED"
72 } else if active == "failed" {
73 status = "FAILED"
74 } else {
75 status = strings.ToUpper(active)
76 }
77
78 // Try to get PID for running services
79 var pid *int
80 if status == "RUNNING" {
81 pid = getServicePIDLinux(serviceName)
82 }
83
84 services = append(services, Service{
85 Name: serviceName,
86 Status: status,
87 Description: description,
88 PID: pid,
89 })
90 }
91
92 return services, nil
93}

95// getServicePIDLinux attempts to get the main PID for a service
96func getServicePIDLinux(serviceName string) *int {

98 output, err := cmd.Output()
