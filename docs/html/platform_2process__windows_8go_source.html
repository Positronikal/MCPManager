38 )

43 }
44 defer procCloseHandle.Call(handle)

51 return false
52 }
53
54 // If exit code is STILL_ACTIVE, the process is still running
55 isRunning := exitCode == STILL_ACTIVE
56 fmt.Printf("[ProcessManager] PID %d is running: %v (exitCode=%d)\n", pid, isRunning, exitCode)
