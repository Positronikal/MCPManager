88// Stop terminates a process by its ID
89func (pm *DefaultProcessManager) Stop(pid int, graceful bool, timeout int) error {
90 fmt.Printf("[ProcessManager] Stop called: pid=%d, graceful=%v, timeout=%d\n", pid, graceful, timeout)
91
92 // Find the process
93 process, err := os.FindProcess(pid)
94 if err != nil {
95 fmt.Printf("[ProcessManager] os.FindProcess failed: pid=%d, error=%v\n", pid, err)
96 return fmt.Errorf("process not found: %w", err)
97 }
98 fmt.Printf("[ProcessManager] Process found: pid=%d\n", pid)
99
100 if graceful {
101 fmt.Printf("[ProcessManager] Attempting graceful shutdown: pid=%d\n", pid)
102 // Try graceful shutdown first
103 if err := pm.sendTermSignal(process); err != nil {
104 fmt.Printf("[ProcessManager] Graceful termination signal failed: pid=%d, error=%v\n", pid, err)
105 // If signal failed, process might already be dead
106 if !pm.IsRunning(pid) {
107 fmt.Printf("[ProcessManager] Process already dead after signal failure: pid=%d\n", pid)
108 return nil
109 }
110 // Signal failed but process still running, fall through to force kill
111 } else {
112 fmt.Printf("[ProcessManager] Graceful termination signal sent: pid=%d, waiting for exit...\n", pid)
113
114 // Wait for process to exit
115 done := make(chan bool)
116 go func() {
117 // Poll for process exit
118 for i := 0; i < timeout*10; i++ {
119 if !pm.IsRunning(pid) {
120 fmt.Printf("[ProcessManager] Process exited gracefully: pid=%d\n", pid)
121 done <- true
122 return
123 }
124 time.Sleep(100 * time.Millisecond)
125 }
126 fmt.Printf("[ProcessManager] Graceful shutdown timed out: pid=%d\n", pid)
127 done <- false
128 }()
129
130 if <-done {
131 // Process exited gracefully
132 return nil
133 }
134
135 // Graceful shutdown timed out, fall through to force kill
136 fmt.Printf("[ProcessManager] Proceeding to force kill: pid=%d\n", pid)
137 }
138 }
139
140 // Force kill
141 fmt.Printf("[ProcessManager] Attempting force kill: pid=%d\n", pid)
142 if err := process.Kill(); err != nil {
143 fmt.Printf("[ProcessManager] Force kill failed: pid=%d, error=%v\n", pid, err)
144 // Check if process is actually dead
145 if !pm.IsRunning(pid) {
146 fmt.Printf("[ProcessManager] Process is dead despite kill error: pid=%d\n", pid)
147 return nil
148 }
149 return fmt.Errorf("failed to kill process %d: %w", pid, err)
150 }
151
152 fmt.Printf("[ProcessManager] Force kill succeeded: pid=%d\n", pid)
153 return nil
154}

156// sendTermSignal sends a termination signal to the process
157func (pm *DefaultProcessManager) sendTermSignal(process *os.Process) error {
158 if runtime.GOOS == "windows" {
159 // Windows doesn't support SIGTERM, use Kill
160 return process.Kill()
161 }
162 // Unix systems: send SIGTERM
163 return process.Signal(syscall.SIGTERM)
