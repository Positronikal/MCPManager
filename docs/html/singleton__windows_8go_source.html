25// WindowsSingleInstance implements SingleInstance for Windows using named mutex
26type WindowsSingleInstance struct {
27 mutexHandle syscall.Handle
28 mutexName string
29 windowTitle string
30}

32// NewSingleInstance creates a new Windows single instance enforcer
33func NewSingleInstance(appName string, windowTitle string) SingleInstance {
34 return &WindowsSingleInstance{
35 mutexName: fmt.Sprintf("Global\\%s", appName),
36 windowTitle: windowTitle,
37 }
38}

40// Acquire attempts to acquire the single instance lock
41func (w *WindowsSingleInstance) Acquire() (bool, error) {
42 mutexNamePtr, err := syscall.UTF16PtrFromString(w.mutexName)
43 if err != nil {
44 return false, fmt.Errorf("failed to convert mutex name to UTF16: %w", err)
45 }
46
47 // Try to create the mutex
48 ret, _, err := procCreateMutex.Call(
49 0, // lpMutexAttributes (NULL)
50 0, // bInitialOwner (FALSE)
51 uintptr(unsafe.Pointer(mutexNamePtr)), // lpName
52 )
53
54 if ret == 0 {
55 return false, fmt.Errorf("failed to create mutex: %w", err)
56 }
57
58 w.mutexHandle = syscall.Handle(ret)
59
60 // Check if mutex already existed
61 if err == syscall.Errno(ERROR_ALREADY_EXISTS) {
62 // Another instance is running - bring it to foreground
63 w.bringExistingWindowToFront()
64 // Close the mutex handle since we won't be using it
65 procCloseHandle.Call(uintptr(w.mutexHandle))
66 w.mutexHandle = 0
67 return false, nil
68 }
69
70 // We successfully acquired the mutex (first instance)
71 return true, nil
72}
73
74// Release releases the single instance lock
75func (w *WindowsSingleInstance) Release() error {
76 if w.mutexHandle == 0 {
77 return nil // Already released or never acquired
78 }
79
80 // Release the mutex
81 ret, _, err := procReleaseMutex.Call(uintptr(w.mutexHandle))
82 if ret == 0 {
83 return fmt.Errorf("failed to release mutex: %w", err)
84 }
85
86 // Close the mutex handle
87 ret, _, err = procCloseHandle.Call(uintptr(w.mutexHandle))
88 if ret == 0 {
89 return fmt.Errorf("failed to close mutex handle: %w", err)
90 }
91
92 w.mutexHandle = 0
93 return nil
94}

96// bringExistingWindowToFront finds the existing application window and brings it to foreground
97func (w *WindowsSingleInstance) bringExistingWindowToFront() {
98 if w.windowTitle == "" {
99 return
100 }
101
102 windowTitlePtr, err := syscall.UTF16PtrFromString(w.windowTitle)
103 if err != nil {
104 return
105 }
106
107 // Find the window by title
108 hwnd, _, _ := procFindWindow.Call(
109 0, // lpClassName (NULL - search by title only)
110 uintptr(unsafe.Pointer(windowTitlePtr)),
111 )
112
113 if hwnd == 0 {
114 // Window not found - might have different title or not created yet
115 return
116 }
117
118 // Restore the window if it's minimized
119 procShowWindow.Call(hwnd, SW_RESTORE)
120
121 // Bring the window to foreground
122 procSetForegroundWindow.Call(hwnd)
123}
